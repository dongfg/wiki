Git基本操作

初始化

git init

提交文件

touch README.MD
edit README.MD
git add README.MD
git commit -m "add readme"
git push origin master

创建/切换分支

git checkout -b v1.2.0

合并(Merge)分支

git checkout

删除分支

git branch -D v1.2.0
git push origin --delete v1.2.0
